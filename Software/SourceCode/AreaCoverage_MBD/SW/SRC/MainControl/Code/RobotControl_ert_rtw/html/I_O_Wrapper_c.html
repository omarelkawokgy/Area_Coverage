114void HMC5883L_CompassRead(int16_t *Temp_angle)



115{



116/* CONNECTION PINS FOR THIS ARE A5 PIN (I2C PIN) >> SCL ..... A4 PIN (I2C PIN) >> SCA */



117    //raw data so its not scaled



118  MagnetometerRaw raw = Compass.ReadRawAxis();



119  



120  MagnetometerScaled scaled = Compass.ReadScaledAxis();



121



122  float heading = atan2(raw.YAxis, raw.XAxis);



123



124  if (heading < 0)



125  {



126    heading += 2 * PI;



127  }



128



129  //convert radians to degrees



130  *Temp_angle = (int16_T)(heading * 180 / M_PI);



131}

133void ULSL_Hndler_Outputs_wrapper(const uint8_t ULSL_Pin, uint16_t *ULSL_cm)



134{



135  pinMode(ULSL_Pin, OUTPUT);



136  



137  digitalWrite(ULSL_Pin, LOW);



138  delayMicroseconds(3);



139  digitalWrite(ULSL_Pin, HIGH);



140  delayMicroseconds(5);



141  digitalWrite(ULSL_Pin, LOW);



142



143  /* read the signal from the sensor*/



144  pinMode(ULSL_Pin, INPUT);



145



146  uint32_t duration = pulseIn(ULSL_Pin, HIGH, PULSEIN_TIMEOUT);



147



148  /* convert micro-seconds to centimeters*/



149  *ULSL_cm = (uint16_t)MICRO_SECONDS_TO_CM(duration);



150}
