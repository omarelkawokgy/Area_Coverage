123void HMC5883L_CompassRead(int16_t *Temp_angle)



124{



125/* CONNECTION PINS FOR THIS ARE A5 PIN (I2C PIN) >> SCL ..... A4 PIN (I2C PIN) >> SCA */



126    //raw data so its not scaled



127#ifdef VARIANT_COMPASS_CONNECTED



128  MagnetometerRaw raw = Compass.ReadRawAxis();



129



130  MagnetometerScaled scaled = Compass.ReadScaledAxis();



131



132  float heading = atan2(raw.YAxis, raw.XAxis);



133



134  if (heading < 0)



135  {



136    heading += 2 * PI;



137  }



138



139  //convert radians to degrees



140  *Temp_angle = (int16_T)(heading * 180 / M_PI);



141#else



142  *Temp_angle = (int16_T)(0);



143#endif



144}

146void ULSL_Hndler_Outputs_wrapper(const uint8_t ULSL_Pin, uint16_t *ULSL_cm)



147{



148  pinMode(ULSL_Pin, OUTPUT);



149  



150  digitalWrite(ULSL_Pin, LOW);



151  delayMicroseconds(3);



152  digitalWrite(ULSL_Pin, HIGH);



153  delayMicroseconds(5);



154  digitalWrite(ULSL_Pin, LOW);



155



156  /* read the signal from the sensor*/



157  pinMode(ULSL_Pin, INPUT);



158



159  uint32_t duration = pulseIn(ULSL_Pin, HIGH, PULSEIN_TIMEOUT);



160



161  /* convert micro-seconds to centimeters*/



162  *ULSL_cm = (uint16_t)MICRO_SECONDS_TO_CM(duration);



163}
